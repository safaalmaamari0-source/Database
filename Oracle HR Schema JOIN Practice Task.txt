---Employee & Department Reports---

1. Employee ID, full name, and department name
   (INNER JOIN: employees WITH departments only)
SELECT E.EMPLOYEE_ID,
       E.FIRST_NAME || ' ' || E.LAST_NAME AS Full_Name,
       D.DEPARTMENT_NAME
FROM EMPLOYEES E
INNER JOIN DEPARTMENTS D
ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
ORDER BY E.EMPLOYEE_ID;

2. Employee full name, job title, and salary
   (INNER JOIN: employees WITH valid jobs only)
SELECT E.FIRST_NAME || ' ' || E.LAST_NAME AS Full_Name,
       J.JOB_TITLE,
       E.SALARY
FROM EMPLOYEES E
INNER JOIN JOBS J
ON E.JOB_ID = J.JOB_ID
ORDER BY E.SALARY DESC;

3. Employee, department, and department city
SELECT E.FIRST_NAME || ' ' || E.LAST_NAME AS Full_Name,
       D.DEPARTMENT_NAME,
       L.CITY
FROM EMPLOYEES E
INNER JOIN DEPARTMENTS D ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
INNER JOIN LOCATIONS   L ON D.LOCATION_ID   = L.LOCATION_ID
ORDER BY D.DEPARTMENT_NAME;

4. Employee, department, country, and region
SELECT E.FIRST_NAME || ' ' || E.LAST_NAME AS Full_Name,
       D.DEPARTMENT_NAME,
       C.COUNTRY_NAME,
       R.REGION_NAME
FROM EMPLOYEES  E
INNER JOIN DEPARTMENTS D ON E.DEPARTMENT_ID  = D.DEPARTMENT_ID
INNER JOIN LOCATIONS   L ON D.LOCATION_ID    = L.LOCATION_ID
INNER JOIN COUNTRIES   C ON L.COUNTRY_ID     = C.COUNTRY_ID
INNER JOIN REGIONS     R ON C.REGION_ID      = R.REGION_ID
ORDER BY R.REGION_NAME, C.COUNTRY_NAME;

---Organization Structure Analysis---

5. SELECT E.EMPLOYEE_ID,
       E.FIRST_NAME || ' ' || E.LAST_NAME AS Full_Name,
       NVL(D.DEPARTMENT_NAME, 'No Department Assigned') AS Department
FROM EMPLOYEES   E
LEFT JOIN DEPARTMENTS D ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
ORDER BY E.EMPLOYEE_ID;

6. Complete department list INCLUDING departments without employees
  (RIGHT JOIN: keeps all departments even if no employees)
SELECT D.DEPARTMENT_ID,
       D.DEPARTMENT_NAME,
       COUNT(E.EMPLOYEE_ID) AS Total_Employees
FROM EMPLOYEES   E
RIGHT JOIN DEPARTMENTS D ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
GROUP BY D.DEPARTMENT_ID, D.DEPARTMENT_NAME
ORDER BY D.DEPARTMENT_ID;

 7. Departments that currently have NO employees assigned
SELECT D.DEPARTMENT_ID,
       D.DEPARTMENT_NAME
FROM EMPLOYEES   E
RIGHT JOIN DEPARTMENTS D ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
WHERE E.EMPLOYEE_ID IS NULL
ORDER BY D.DEPARTMENT_ID;

---Management Hierarchy Analysis---

 8. Employee name and their direct manager name
   (SELF JOIN on EMPLOYEES table)
SELECT E.FIRST_NAME  || ' ' || E.LAST_NAME  AS Employee_Name,
       M.FIRST_NAME  || ' ' || M.LAST_NAME  AS Manager_Name
FROM EMPLOYEES E
INNER JOIN EMPLOYEES M ON E.MANAGER_ID = M.EMPLOYEE_ID
ORDER BY Manager_Name, Employee_Name;

 9. Employees WITHOUT managers
  Explanation: These are top-level executives (e.g. the President)
  who have no one above them in the hierarchy.
  Their MANAGER_ID column is NULL.

SELECT EMPLOYEE_ID,
       FIRST_NAME || ' ' || LAST_NAME AS Full_Name,
       JOB_ID,
       'No manager - Top level executive' AS Explanation
FROM EMPLOYEES
WHERE MANAGER_ID IS NULL;

---Advanced Business Analysis---

 10. Employees earning MORE than their managers

SELECT E.FIRST_NAME || ' ' || E.LAST_NAME AS Employee_Name,
       E.SALARY                            AS Employee_Salary,
       M.FIRST_NAME || ' ' || M.LAST_NAME AS Manager_Name,
       M.SALARY                            AS Manager_Salary
FROM EMPLOYEES E
INNER JOIN EMPLOYEES M ON E.MANAGER_ID = M.EMPLOYEE_ID
WHERE E.SALARY > M.SALARY
ORDER BY E.SALARY DESC;

11. Departments where the manager works WITHIN the same department

SELECT D.DEPARTMENT_NAME,
       E.FIRST_NAME || ' ' || E.LAST_NAME AS Manager_Name,
       E.DEPARTMENT_ID                     AS Manager_Dept,
       D.DEPARTMENT_ID                     AS Department
FROM DEPARTMENTS D
INNER JOIN EMPLOYEES E ON D.MANAGER_ID = E.EMPLOYEE_ID
WHERE E.DEPARTMENT_ID = D.DEPARTMENT_ID
ORDER BY D.DEPARTMENT_NAME;

---Executive Dashboard Reports---

12. Employee, job title, department, city, and country

SELECT E.FIRST_NAME || ' ' || E.LAST_NAME AS Employee_Name,
       J.JOB_TITLE,
       D.DEPARTMENT_NAME,
       L.CITY,
       C.COUNTRY_NAME
FROM EMPLOYEES  E
INNER JOIN JOBS        J ON E.JOB_ID        = J.JOB_ID
INNER JOIN DEPARTMENTS D ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
INNER JOIN LOCATIONS   L ON D.LOCATION_ID   = L.LOCATION_ID
INNER JOIN COUNTRIES   C ON L.COUNTRY_ID    = C.COUNTRY_ID
ORDER BY D.DEPARTMENT_NAME, E.LAST_NAME;

13. Department, manager, city, and country
  (LEFT JOIN: includes departments WITHOUT managers)

SELECT D.DEPARTMENT_NAME,
       NVL(E.FIRST_NAME || ' ' || E.LAST_NAME, 'No Manager') AS Manager_Name,
       L.CITY,
       C.COUNTRY_NAME
FROM DEPARTMENTS D
LEFT JOIN EMPLOYEES  E ON D.MANAGER_ID  = E.EMPLOYEE_ID
LEFT JOIN LOCATIONS  L ON D.LOCATION_ID = L.LOCATION_ID
LEFT JOIN COUNTRIES  C ON L.COUNTRY_ID  = C.COUNTRY_ID
ORDER BY D.DEPARTMENT_NAME;

14. Employees working in the SAME department as their manager

SELECT E.FIRST_NAME || ' ' || E.LAST_NAME AS Employee_Name,
       E.DEPARTMENT_ID                     AS Employee_Dept,
       M.FIRST_NAME || ' ' || M.LAST_NAME AS Manager_Name,
       M.DEPARTMENT_ID                     AS Manager_Dept
FROM EMPLOYEES E
INNER JOIN EMPLOYEES M ON E.MANAGER_ID = M.EMPLOYEE_ID
WHERE E.DEPARTMENT_ID = M.DEPARTMENT_ID
ORDER BY E.DEPARTMENT_ID;

15. Employees whose managers work in DIFFERENT departments

SELECT E.FIRST_NAME || ' ' || E.LAST_NAME AS Employee_Name,
       E.DEPARTMENT_ID                     AS Employee_Dept,
       M.FIRST_NAME || ' ' || M.LAST_NAME AS Manager_Name,
       M.DEPARTMENT_ID                     AS Manager_Dept
FROM EMPLOYEES E
INNER JOIN EMPLOYEES M ON E.MANAGER_ID = M.EMPLOYEE_ID
WHERE E.DEPARTMENT_ID != M.DEPARTMENT_ID
ORDER BY E.DEPARTMENT_ID;

16. Distinct countries containing company departments

SELECT DISTINCT C.COUNTRY_NAME,
                R.REGION_NAME
FROM DEPARTMENTS D
INNER JOIN LOCATIONS L ON D.LOCATION_ID = L.LOCATION_ID
INNER JOIN COUNTRIES C ON L.COUNTRY_ID  = C.COUNTRY_ID
INNER JOIN REGIONS   R ON C.REGION_ID   = R.REGION_ID
ORDER BY R.REGION_NAME, C.COUNTRY_NAME;
