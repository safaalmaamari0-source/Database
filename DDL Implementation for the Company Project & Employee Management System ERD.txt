CREATE TABLE Employee(
SSN NUMBER(9) PRIMARY KEY,
FName VARCHAR(20) NOT NULL,
LName VARCHAR(20) NOT NULL,
BD DATE,
gender VARCHAR2(10)
);

CREATE TABLE Departments(
DNum NUMBER(9) PRIMARY KEY,
DName VARCHAR(20) NOT NULL,
Hiring_date DATE,
SSN NUMBER(9),
CONSTRAINT fk_dept_mgr FOREIGN KEY (SSN) REFERENCES Employee(SSN)
);

CREATE TABLE DEPT_LOCATIONS (
Dnumber INT,
Dlocation VARCHAR(50),
PRIMARY KEY (Dnumber, Dlocation),
CONSTRAINT fk_dept_loc FOREIGN KEY (Dnumber) REFERENCES Departments(Dnum) ON DELETE CASCADE
);

CREATE TABLE PROJECT (
Pnumber INT PRIMARY KEY,
Pname VARCHAR(50) NOT NULL UNIQUE,
Plocation VARCHAR(50),
City VARCHAR(50),
Dnum NUMBER,
CONSTRAINT fk_project_dept FOREIGN KEY (DNUM) REFERENCES Departments(Dnum) ON DELETE SET NULL
);


CREATE TABLE WORKS_ON (
SSN NUMBER(9),
Pnumber INT,
Hours DECIMAL(5,1),
PRIMARY KEY (SSN, Pnumber), CONSTRAINT fk_work_emp FOREIGN KEY (SSN) REFERENCES Employee(SSN) ON DELETE CASCADE, 
CONSTRAINT fk_work_proj FOREIGN KEY (Pnumber) REFERENCES PROJECT(Pnumber) ON DELETE CASCADE
);

CREATE TABLE DEPENDENT (
SSN NUMBER(9),
Dependent_name VARCHAR(50),
Bday DATE,
PRIMARY KEY (SSN, Dependent_name), CONSTRAINT fk_dependent_emp FOREIGN KEY (SSN) REFERENCES Employee(SSN) ON DELETE CASCADE
);
